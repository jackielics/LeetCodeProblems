# Intuition
DFS, backtrack judge if panlindrome. do not use traditional backtrack in Three var
2-DP

# Approach

# Complexity
- Time complexity:
$$O(2^n)$$

- Space complexity:
$$O(n)$$

# Code
```python
class Solution:
    def partition0(self, s: str) -> List[List[str]]:
        def is_panli(i, j):
            if i >= j:
                return True
            return is_panli(i + 1, j - 1) if s[i] == s[j] else False

        res = []
        part = []
        def dfs(i: int):
            if i >= len(s):
                res.append(part.copy())
                return
            for j in range(i, len(s)):
                if is_panli(i, j):
                    part.append(s[i:j+1])
                    dfs(j + 1)
                    part.pop()
        dfs(0)
        return res

    def partition1(self, s: str) -> List[List[str]]:
        '''
        2-DP
        '''
        n = len(s)
        is_panli = [[True] * n for _ in range(n)]

        for start in range(n - 1, -1, -1):
            for end in range(start + 1, n): # upper triangle
                is_panli[start][end] = (s[start] == s[end]) and is_panli[start + 1][end - 1]

        res = []
        part = []
        def dfs(i: int):
            if i >= len(s):
                res.append(part.copy())
                return
            for j in range(i, len(s)):
                if is_panli[i][j]:
                    part.append(s[i:j+1])
                    dfs(j + 1)
                    part.pop()
        dfs(0)
        return res
```
