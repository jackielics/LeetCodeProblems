# Intuition
DFS, backtrack judge if panlindrome. do not use traditional backtrack in Three var
2-DP

# Approach

# Complexity
- Time complexity:
$$O(2^n)$$

- Space complexity:
$$O(n)$$

# Code
```python
class Solution:
    def partition0(self, s: str) -> List[List[str]]:
        def is_panli(i, j):
            if i >= j:
                return True
            return s[i] == s[j] and is_panli(i + 1, j - 1)

        res = []
        part = []
        def dfs(i: int):
            if i >= len(s):
                res.append(part.copy())
                return
            for j in range(i, len(s)):
                if is_panli(i, j):
                    part.append(s[i:j+1])
                    dfs(j + 1)
                    part.pop()
        dfs(0)
        return res

    def partition(self, s: str) -> List[List[str]]:
        '''
        use 2-DP to initialize panlidrome matrix
        '''
        n = len(s)
        panlin_matrix = [[True] * n for _ in range(n)]

        for i in range(n - 1, -1, -1):
            for j in range(i + 1, n): # upper triangle
                panlin_matrix[i][j] = (s[i] == s[j]) and panlin_matrix[i + 1][j - 1]

        res = []
        part = []
        def dfs(i: int):
            if i >= len(s):
                res.append(part.copy())
                return
            for j in range(i, len(s)):
                if panlin_matrix[i][j]:
                    part.append(s[i:j+1])
                    dfs(j + 1)
                    part.pop()
        dfs(0)
        return res
```
