# Code
```python
class Solution:
	def isValid1(self, s: str) -> bool:
		'''
		Master's Solution: use DICT to store pairs
		Time Complexity: O(n)
		Space Complexity: O(n)
		'''
		pairs = {
			'(': ')',
			'[': ']',
			'{': '}'
		}
		stack = []
		for c in s:
			if c in pairs:
				stack.append(c)
			elif not stack or c != stack[stack.pop()]:
				return False
	
		return len(stack) == 0
```
