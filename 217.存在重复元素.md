# Intuition
hashset, sort, count()

# Approach


# Complexity
- Time complexity:
$$O(n)$$

- Space complexity:
$$O(n)$$

# Code
```python
class Solution:
	def containsDuplicate0(self, nums: List[int]) -> bool:
		'''
		Master's solution1: use set
		Time:O(n)
		Space:O(n)
		'''
		return len(nums) != len(set(nums))


	def containsDuplicate1(self, nums: List[int]) -> bool:
		'''
		Master's solution3: use HashSet
		Time:O(n)
		Space:O(n)
		'''
		hashset = set()
		for n in nums:
			if n in hashset:
				return True
			hashset.add(n)
		return False
	
	def containsDuplicate2(self, nums: List[int]) -> bool:
		'''
		Master's solution2:, sort
		Time:O(nlogn)
		Space:O(1)
		'''
		nums.sort()
		for i in range(len(nums) - 1):
			if nums[i] == nums[i+1]:
				return True
		return False

	def containsDuplicate3(self, nums: List[int]) -> bool:
		'''
		Master's solution2:, sort
		Time:O(nlogn)
		Space:O(1)
		'''
		for i in nums:
			if nums.count(i) > 1:
				return True
			
		return False
```
