$$O(2^n)$$

- Space complexity:
$$O(n)$$

class Solution:
    def letterCombinations0(self, digits: str) -> List[str]:
        mapping_dic = {
            '2': ['a', 'b', 'c'],
            '3': ['d', 'e', 'f'],
            '4': ['g', 'h', 'i'],
            '5': ['j', 'k', 'l'],
            '6': ['m', 'n', 'o'],
            '7': ['p', 'q', 'r', 's'],
            '8': ['t', 'u', 'v'],
            '9': ['w', 'x', 'y', 'z']
        }

        global res
        res = []

        def backtrack(cur: list, i: int):
            if i >= len(digits):
                global res
                res = cur.copy()
                # res.append(cur)
                return
            
            newCur = []
            if not cur:
                for x in mapping_dic[digits[i]]:
                    cur.append(x)
                backtrack(cur, i + 1)
            else:
                for x in cur:
                    for y in mapping_dic[digits[i]]:    
                        newCur.append(x + y)
                backtrack(newCur, i + 1)
        

        backtrack([], 0)
        return res
    
    def letterCombinations(self, digits: str) -> List[str]:
        '''
        backtrack cur STRING
        '''
        digitToChar = {
            "2": "abc",
            "3": "def",
            "4": "ghi",
            "5": "jkl",
            "6": "mno",
            "7": "qprs",
            "8": "tuv",
            "9": "wxyz"
        }
        res = []
        def backtrack(curStr: str, i: int):
            if len(curStr) == len(digits):
                res.append(curStr)
                return
            
            for c in digitToChar[digits[i]]:
                backtrack(curStr + c, i + 1)

        if digits:
            backtrack("", 0)
        return res
